## Install

[npm][]:

```sh
npm install koa-cors-gate
```

```sh
yarn add koa-cors-gate
```


## Usage

```js
const KoaCorsGate = require('koa-cors-gate');

const koaCorsGate = new KoaCorsGate();

console.log(koaCorsGate.renderName());
// script
```
